% inserting figures
\newcommand{\insertfigure}[3]{ % Filename, Caption, Width percent of textwidth
	\begin{figure}[h!]
		\begin{center}
			\includegraphics[width=#3\textwidth]{#1}
		\end{center}
		\vspace{-0.4cm}
		\caption{#2}
		\label{fig:#1}
	\end{figure}
}

\newcommand{\insertsubfigures}[7]{ % caption, imageleft, captionleft, widthleft, imageright, captionright, widthright
\begin{figure}[h!]
\centering
\begin{subfigure}{.5\textwidth}
  \centering
  \includegraphics[width=#4\linewidth]{#2}
  \caption{#3}
  \label{fig:#2}
\end{subfigure}%
\begin{subfigure}{.5\textwidth}
  \centering
  \includegraphics[width=#7\linewidth]{#5}
  \caption{#6}
  \label{fig:#5}
\end{subfigure}
\caption{#1}
\label{fig:#2#5}
\end{figure}
}

\newcommand{\insertsubfigurespdf}[7]{ % caption, imageleft, captionleft, widthleft, imageright, captionright, widthright
\begin{figure}[h!]
\centering
\begin{subfigure}{.5\textwidth}
  \centering
	\def\svgwidth{#4\textwidth}
	\input{figures/#2.pdf_tex}
  \caption{#3}
  \label{fig:#2}
\end{subfigure}%
\begin{subfigure}{.5\textwidth}
  \centering
	\def\svgwidth{#7\textwidth}
	\input{figures/#5.pdf_tex}
  \caption{#6}
  \label{fig:#5}
\end{subfigure}
\caption{#1}
\label{fig:#2#5}
\end{figure}
}

% inserting inkskape figures
\newcommand{\insertpdffigure}[3]{ % Filename, Caption, Label, Width percent of textwidth

		\end{center}

		\label{fig:#1}
	\end{figure}
}
